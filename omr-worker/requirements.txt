# OMR worker Python dependencies. Install into a venv (two steps, see README.md):
#   pip install -r requirements.txt
#   pip install --no-deps -r requirements-oemer.txt
# NOTE: poppler-utils (the pdftoppm binary) is an OS package, not pip. Install it with
#   sudo apt-get install -y poppler-utils   (Linux)  /  brew install poppler  (macOS)
# See README.md.

# Fallback OMR engine, tried only when oemer produces nothing. Pulls a torch/transformers
# ML stack and requires a modern numpy (>=2.2.6).
homr

# Pin the numpy band both engines tolerate. oemer 0.1.8 runs on numpy 2.x but a future
# numpy 3.0 could re-break it the way numpy 1.24 broke 0.1.5 (and the failure is silent:
# the worker just falls back to homr and collapses the grand staff). Lower bound matches
# homr's floor.
numpy>=2.2.6,<3

# oemer's runtime dependencies, listed here so the second (--no-deps) install of oemer
# still has everything it imports. oemer itself is pinned in requirements-oemer.txt and
# installed with --no-deps because it declares onnxruntime-gpu, which has no ARM/macOS
# wheel; we run CPU inference through onnxruntime instead (same `onnxruntime` module).
onnxruntime
opencv-python-headless>=4.5.3.56
matplotlib
pillow
scipy
scikit-learn>=1.2
typing-extensions

# S3-compatible client used to talk to Cloudflare R2.
boto3

# Visual-diff PDF referee (omr-worker/referee.py, ensemble Slice 6). Engraves each engine's
# candidate back to an image to arbitrate isolated-notehead pitch disputes. The module is
# guarded (REFEREE_AVAILABLE) so it degrades cleanly if these are absent, and is NOT yet
# wired into worker.py/reconcile.py, so adding these is inert until a later slice flips a gate.
# IMPORTANT INSTALL CONSTRAINTS (verified in Slices 5/6):
#   - verovio's linux-x86_64 (manylinux) wheel exists for Python >= 3.11 ONLY. The cx33
#     worker venv MUST run Python >= 3.11 or pip falls back to building from source (needs
#     swig, fails). The current prod worker venv does not yet meet this; the cx33 install +
#     Python bump is a SEPARATE infra task (do not assume these import on the box yet).
#   - cairosvg dlopens the system libcairo at runtime via cairocffi (pure-Python, no bundled
#     cairo), so the host also needs the OS package libcairo2 (apt-get install libcairo2).
#     Verovio has no PNG output and Pillow cannot read SVG, so cairosvg is required for SVG->PNG.
verovio          # MusicXML -> SVG engraver; LGPL/GPL-3.0 (compatible with the GPL-3.0 Clarity dep)
cairosvg         # SVG -> PNG rasterizer (needs system libcairo2)
